Example

#include "LSort.h"
#include "LSystem.h"

int main(){
    ui4 i, k = 5, theItems[11] = { 34, 12, 6, 9, 33, 11, 4, 3, 11, 23, 26 };

    /* print array */
    for (i = 0; i < 11; ++i) LSystem_Print("%lu ", theItems[i]);
    LSystem_Print("\n");

    /* sort array */
    LSort_RecQuick(theItems, 11, LType_UI4);

    /* print sorted array */
    for (i = 0; i < 11; ++i) LDebug_Print("%lu ", theItems[i]);
    LSystem_Print("\n");

    return 0;
}

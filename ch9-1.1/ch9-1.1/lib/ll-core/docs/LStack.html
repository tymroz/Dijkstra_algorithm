Introduction

The component LStack provides support for maintaining a last-in-first-out (LIFO) stack data structure.

Example

#include "LStack.h"
#include "LException.h"
#include "LDebug.h"

int main() {
    ui4         i;
    LStack*     theStack = NULL;
    LException* theExcep;
    
    Try {
        /* create new stack */
        theStack = LStack_New(LType_UI4);
    
        /* adds values to the stack */
        for (i=100; i>0; --i)
            LStack_Push(theStack, &i);
    
        /* extracts all values from the stack */
        while (!LStack_IsEmpty(theStack)) {
            LStack_Top(theStack, &i);
            LStack_Pop(theStack);
            LDebug_Print("Extracted value: %lu\n", i);
        }
    }
    
    Catch(theExcep) {
        LException_Dump(theExcep);
    }
    
    if (theStack != NULL) LStack_Delete(&theStack);
    
    return 0;
}
